UserController.java


package com.controller;

import java.util.List;

import javax.servlet.http.HttpServletRequest;

import org.springframework.beans.factory.annotation.Autowired;

import org.springframework.stereotype.Controller;

import org.springframework.ui.Model;

import org.springframework.web.bind.annotation.RequestMapping;

import com.dao.UserDAO;

import com.entity.Admin;

import com.entity.User;

import com.service.UserService;

@Controller

public class UserController {

@Autowired

private UserService userService;

@Autowired

private UserDAO userDAO;

//登录

    @RequestMapping("/login")

public String login(HttpServletRequest req){

if(null!=userService.login(req.getParameter("username"),req.getParameter("password"))) {

User[] list = userDAO.search();

req.setAttribute("alluserList",list);

return "redirect:select";

}

else

return "login";

}

//查询列表信息

@RequestMapping("/select")

public String select(Model model) {

model.addAttribute("alluserList", userService.search());

return "alluserList";

}

//模糊查询

@RequestMapping("/selectByLike")

public String select(String column,String keyword,Model model) {

column=column==null?"":column;

keyword=keyword==null?"":keyword;

model.addAttribute("column",column );

model.addAttribute("keyword",keyword );

List<User> list = userDAO.selectByLike(column,keyword);

model.addAttribute("alluserList",list);

return "alluserList";

}

//添加信息

@RequestMapping("/insert")

public String insertUser(HttpServletRequest req,User user,Model model){

user.setId(user.getId());

user.setName(user.getName());

user.setBirthday(user.getBirthday());

user.setSex(user.getSex());

user.setProfession(user.getProfession());

user.setAdress(user.getAdress());

user.setMobile_phone(user.getMobile_phone());

System.out.println(user.toString());

String insert = userService.insertUser(user);

req.setAttribute("insert",insert);

model.addAttribute("alluserList", userService.search());

return "alluserList";

}

//修改

@RequestMapping("/update")

public String updateUser(HttpServletRequest req,User user){

String update = userService.updateUser(user);

req.setAttribute("update",update);

req.setAttribute("user",user);

return "update";

}

//edit

@RequestMapping("/edit")

public String edit(HttpServletRequest req,int id){

req.setAttribute("user",userService.get(id));

return "update";

}

//edit1

@RequestMapping("/edit1")

public String edit1(HttpServletRequest req,int id){

req.setAttribute("user",userService.get(id));

return "delete";

}

//删除

@RequestMapping("/delete")

public String deleteUser(HttpServletRequest req,User user) {

user.setId(user.getId());

String delete = userService.deleteUser(user);

req.setAttribute("delete", delete);

req.setAttribute("user",user);

return "delete";

}

}
